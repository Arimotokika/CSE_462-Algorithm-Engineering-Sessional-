% ── SLIDE: Problem Definition ─────────────────────────────────────────
\begin{frame}{Problem Definition}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\small\bfseries, anchor=west] at (0.2,0.225)
      {\faShield*\quad Weapon-Target Assignment (WTA) Problem};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Optimally allocate weapons to targets to \textbf{minimize residual threat}
  under real-time operational constraints.
  }

  \vspace{0.3cm}

  \begin{tikzpicture}
    \fill[tealaccent, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\small\bfseries, anchor=west] at (0.2,0.225)
      {\faBolt\quad Core Challenge: Stochastic Combat};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Weapon $w_i$ destroys target $t_j$ with kill probability $p_{ij} \in [0,1]$.
  Assignments must be optimized \textbf{under uncertainty}, in real time.
  }

  \vspace{0.3cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\small\bfseries, anchor=west] at (0.2,0.225)
      {\faMicrochip\quad Computational Complexity};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  \begin{itemize}
    \setlength\itemsep{3pt}
    \item[\color{goldaccent}$\blacktriangleright$]
      Proven \textbf{NP-complete} — no polynomial-time exact solution exists
    \item[\color{goldaccent}$\blacktriangleright$]
      Search space grows \textbf{exponentially} with weapons and targets
    \item[\color{goldaccent}$\blacktriangleright$]
      Large instances require \textbf{heuristic or metaheuristic} approaches
  \end{itemize}
  }

\end{frame}

    \item[\color{goldaccent}$\blacktriangleright$]

    \item[\color{goldaccent}$\blacktriangleright$]

  \end{itemize}

  \end{center}

  \vspace{0.15cm}
  {\footnotesize
  \textbf{Integer variable} representing the number of weapons of type $i$ assigned to target $j$
  }

\end{frame}

% ── SLIDE: Objective Functions ────────────────────────────────────────
\begin{frame}{Mathematical Representation: Objective Functions}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faBullseye\quad Goal: Maximize damage (or equivalently, minimize survival)};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  While the goal is to maximize damage, the standard convention is to minimize
  remaining survival value.
  }

  \vspace{0.35cm}

  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\footnotesize\color{navydark}\textbf{Maximizing Total Expected Damage}}
      \vspace{0.2cm}
      {\small
      \[
        \text{Max } Z = \sum_{j=1}^{n} v_j
        \left[ 1 - \prod_{i=1}^{m} (1-p_{ij})^{x_{ij}} \right]
      \]
      }
    \end{column}

    \begin{column}{0.48\textwidth}
      {\footnotesize\color{tealaccent}\textbf{Minimizing Expected Survival Value}}
      \vspace{0.2cm}
      {\small
      \[
        \text{Min } Z = \sum_{j=1}^{n} v_j \prod_{i=1}^{m} (1-p_{ij})^{x_{ij}}
      \]
      }
    \end{column}

  \end{columns}

  \vspace{0.35cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faLightbulb\quad Key Insight};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  $\prod_{i=1}^{m}(1-p_{ij})^{x_{ij}}$ $=$ \textbf{Probability that target $j$ survives ALL assigned weapons}\\[3pt]
  The product captures the \textbf{combined effect} of multiple weapons attacking the same target.
  }

\end{frame}

% ── SLIDE: Constraints ────────────────────────────────────────────────
\begin{frame}{Mathematical Representation: Constraints}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Constraint 1: Resource Availability};
  \end{tikzpicture}

  \vspace{0.2cm}
  \[
    \sum_{j=1}^{n} x_{ij} \leq w_i, \quad \forall i \in W
  \]
  \vspace{-0.2cm}
  {\footnotesize
  \textbf{Meaning:} Total weapons of type $i$ assigned across all targets cannot exceed available quantity $w_i$
  }

  \vspace{0.3cm}

  \begin{tikzpicture}
    \fill[tealaccent, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Constraint 2: Integrality};
  \end{tikzpicture}

  \vspace{0.2cm}
  \[
    x_{ij} \in \{0, 1, 2, \ldots\}
  \]
  \vspace{-0.2cm}
  {\footnotesize
  \textbf{Meaning:} We can only assign \textbf{whole numbers} of weapons (discrete assignments)
  }

  \vspace{0.3cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Complete Formulation Summary};
  \end{tikzpicture}

  \vspace{0.2cm}
  {\footnotesize
  \textbf{Minimize} $\displaystyle\sum_j v_j \prod_i (1-p_{ij})^{x_{ij}}$
  \quad \textbf{s.t.} \quad
  $\displaystyle\sum_j x_{ij} \leq w_i, \quad x_{ij} \in \mathbb{Z}^+$
  }

\end{frame}

% ── SLIDE: WTA Decision Problem ──────────────────────────────────────
\begin{frame}{WTA Decision Problem}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faQuestion\quad Decision Version of WTA};
  \end{tikzpicture}

  \vspace{0.2cm}
  {\footnotesize
  \textbf{Given:}
  \begin{itemize}
    \setlength\itemsep{2pt}
    \item[\color{goldaccent}$\blacktriangleright$] Survival probability $q_{i,j} = 1 - p_{i,j}$
    \item[\color{goldaccent}$\blacktriangleright$] Threshold value $k$
  \end{itemize}
  \textbf{Question:} Does there exist an assignment $x_{i,j} \in \{0,1\}$ such that:
  \[
    D = \sum_{j=1}^{n} V_j \prod_{i=1}^{m} q_{i,j}^{x_{i,j}} \leq k
  \]
  }

  \vspace{0.2cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faLock\quad Subject to Constraints};
  \end{tikzpicture}

  \vspace{0.2cm}
  {\footnotesize
  \[
    \sum_{j=1}^{n} x_{i,j} \leq W_i \quad \text{for } i = 1,\ldots,m
    \qquad x_{i,j} \in \{0,1\}
  \]
  where $x_{i,j} = 1$ if weapon $i$ assigned to target $j$, and $x_{i,j} = 0$ otherwise.\\[3pt]
  $D$ represents \textbf{total expected damage} (weighted survival probability of all targets).
  }

\end{frame}

\begin{frame}{Two Polynomial-Time Reductions}

  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize We demonstrate two polynomial-time reductions:}

  \vspace{0.3cm}

  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      \begin{tikzpicture}

      \end{tikzpicture}
      \vspace{0.15cm}
      {\footnotesize
      \\
      \textbf{SUBSET PRODUCT $\leq_p$ WTA}\\[6pt]

      Reduce FROM a known NP-complete problem TO WTA.
      }
    \end{column}

    \begin{column}{0.48\textwidth}

      Shows connection between problems.
      }
    \end{column}

  \end{columns}

  \vspace{0.5cm}

  \begin{center}
  \begin{tikzpicture}[node distance=3.8cm]

          font=\scriptsize, minimum height=0.65cm, minimum width=2.8cm] (SP)

      {WTA};

% ── SLIDE: Reduction 1 Construction ──────────────────────────────────
\begin{frame}{Reduction 1: SUBSET PRODUCT $\leq_p$ WTA}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {SUBSET PRODUCT Problem (NP-complete)};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  \textbf{Given:} Set $A = \{a_1, a_2, \ldots, a_m\}$ and target product $b$\\
  \textbf{Find:} Subset $X \subseteq A$ where $\prod_{x \in X} a_x = b$
  }

  \vspace{0.25cm}

  \begin{tikzpicture}
    \fill[tealaccent, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {WTA Instance Construction};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Let $P = \prod_{i=1}^{m} a_i$. Construct WTA with:
  \begin{itemize}
    \setlength\itemsep{2pt}
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Targets:} 2 targets with values $V_1 = V_2 = P$
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Weapons:} $m+2$ weapons: $W_1, W_2, \ldots, W_m, W_\alpha, W_\beta$
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Survival probabilities:} $q_{i,1} = q_{i,2} = \frac{1}{a_i}$ for $i=1,\ldots,m$; \quad $q_{\alpha,1} = q_{\alpha,2} = \frac{b}{2P}$; \quad $q_{\beta,1} = q_{\beta,2} = \frac{1}{2b}$
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Threshold:} $k = 1$ (we ask if $D \leq 1$)
  \end{itemize}
  }

  \vspace{0.25cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faClock\quad Time Complexity};
  \end{tikzpicture}

  \vspace{0.1cm}
  {\footnotesize
  Constructs WTA with 2 targets and $m+2$ weapons in $O(m)$ time --- \textbf{Polynomial!}
  }

\end{frame}

% ── SLIDE: Proof of Correctness ───────────────────────────────────────
\begin{frame}{Reduction 1: Proof of Correctness}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Forward Direction ($\Rightarrow$): SUBSET PRODUCT has solution $\Rightarrow$ WTA has solution};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Suppose $X \subseteq \{1,\ldots,m\}$ with $\prod_{x \in X} a_x = b$. Let $Y = \{1,\ldots,m\} \setminus X$. Then:
  \[
    \prod_{i \in X} \frac{1}{a_i} = \frac{1}{b}, \qquad \prod_{y \in Y} \frac{1}{a_y} = \frac{b}{P}
  \]
  \textbf{Assignment:} $\{W_x : x \in X\} \cup \{W_\alpha\} \to T_1$ and $\{W_y : y \in Y\} \cup \{W_\beta\} \to T_2$\\
  \textbf{Total damage:}
  \[
    D = P \cdot \frac{b}{2P} \cdot \frac{1}{b} + P \cdot \frac{1}{2b} \cdot \frac{b}{P} = \frac{1}{2} + \frac{1}{2} = 1 \leq k \quad \checkmark
  \]
  }

  \vspace{0.2cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Backward Direction ($\Leftarrow$): WTA has solution $\Rightarrow$ SUBSET PRODUCT has solution};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Suppose WTA has solution with $D \leq 1$. Let $X', Y'$ be weapons assigned to $T_1, T_2$.\\
  \textbf{Claim:} $W_\alpha$ and $W_\beta$ cannot both target the same target.\\
  \textit{Proof:} If both assigned to $T_1$ (symmetric for $T_2$):
  \[
    D \geq P \cdot \frac{1}{2b} \cdot \frac{b}{2P} \cdot \frac{1}{\prod_{x \in X' \setminus \{\alpha,\beta\}} a_x}
    + \frac{P}{\prod_{y \in Y'} a_y} \geq \frac{1}{4} + 1 > 1 \implies\Leftarrow
  \]
  }

\end{frame}

% ── SLIDE: Backward Direction Continued ──────────────────────────────
\begin{frame}{Reduction 1: Backward Direction (Continued)}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {From the Assignment};
  \end{tikzpicture}

  \vspace{0.1cm}
  {\footnotesize
  With $W_\alpha \to T_1$ and $W_\beta \to T_2$:
  \[
    D = P \cdot \frac{b}{2P} \prod_{x \in X} \frac{1}{a_x} + P \cdot \frac{1}{2b} \prod_{y \in Y} \frac{1}{a_y} \leq 1
  \]
  }

  \vspace{0.1cm}

  \begin{tikzpicture}
    \fill[tealaccent, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Algebraic Derivation};
  \end{tikzpicture}

  \vspace{0.1cm}
  {\footnotesize
  Let $z = \prod_{x \in X} a_x$. Then:
  \[
    \frac{b}{2z} + \frac{z}{2b} \leq 1
  \]
  Multiply by $2bz > 0$:
  \[
    b^2 + z^2 \leq 2bz \implies (b-z)^2 \leq 0
  \]
  Since $(b-z)^2 \geq 0$, we must have $(b-z)^2 = 0$, so:
  \[
    b = z = \prod_{x \in X} a_x
  \]
  }

  \vspace{0.1cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faCheckCircle\quad Conclusion};
  \end{tikzpicture}

  \vspace{0.1cm}
  {\footnotesize
  $X$ is a valid solution to SUBSET PRODUCT! $\checkmark$ \quad
  \textbf{Time Complexity:} $O(m)$ --- Polynomial reduction complete.
  }

\end{frame}

% ── SLIDE: Reduction 2 ────────────────────────────────────────────────
\begin{frame}{Reduction 2: WTA $\leq_p$ Multiple Knapsack}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faBox\quad Multiple Knapsack Construction};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Given WTA with $m$ targets, $n$ weapons, probabilities $\{p_{i,j}\}$:
  \begin{itemize}
    \setlength\itemsep{2pt}
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Knapsacks:} $m$ knapsacks (one per target)
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Capacity:} $C_j = \infty$ (or sufficiently large)
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Items:} $n$ items (one per weapon type)
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Profit:} $v_{ij} = -\log(1-p_{i,j}) = -\log(q_{i,j})$
    \item[\color{goldaccent}$\blacktriangleright$] \textbf{Weight:} $w_{ij} = 0$ (or 1 if weapon limits exist)
  \end{itemize}
  }

  \vspace{0.25cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faLightbulb\quad Objective Transformation (Key Insight)};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  For single target $j$, minimizing $\prod q_{i,j}^{x_{i,j}}$ equals minimizing:
  \[
    \log\left(\prod_{i=1}^{n} q_{i,j}^{x_{i,j}}\right) = \sum_{i=1}^{n} x_{i,j} \log(q_{i,j})
  \]
  Equivalent to \textbf{maximizing}:
  \[
    -\sum_{i=1}^{n} x_{i,j}\log(q_{i,j}) = \sum_{i=1}^{n} x_{i,j} \cdot v_{ij}
  \]
  This is \textbf{exactly the Multiple Knapsack objective!}
  }

\end{frame}

% ── SLIDE: Constraint Mapping ─────────────────────────────────────────
\begin{frame}{Reduction 2: Constraint Mapping \& Conclusion}

  \begin{tikzpicture}
    \fill[navymid, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faProjectDiagram\quad Constraint Mapping};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  \begin{itemize}
    \setlength\itemsep{3pt}
    \item[\color{goldaccent}$\blacktriangleright$]
      \textbf{WTA constraint:} $\sum_{j=1}^{m} x_{i,j} \leq W_i$ (weapon type $i$ used at most $W_i$ times)
    \item[\color{goldaccent}$\blacktriangleright$]
      \textbf{Maps to:} $\sum_{j=1}^{m} y_{ij} \leq 1$ (each item in at most one knapsack) when $W_i = 1$
    \item[\color{goldaccent}$\blacktriangleright$]
      For general $W_i$: Create $W_i$ copies of item $i$
  \end{itemize}
  }

  \vspace{0.25cm}

  \begin{tikzpicture}
    \fill[tealaccent, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {Equivalence};
  \end{tikzpicture}

  \vspace{0.15cm}
  {\footnotesize
  Assignment $x_{i,j}$ optimal for WTA $\Leftrightarrow$ $y_{ij} = x_{i,j}$ optimal for Multiple Knapsack\\
  \textbf{Logarithmic transformation preserves ordering of solutions!}\\
  \textbf{Time:} $O(nm)$ to compute all $v_{ij} = -\log(1-p_{i,j})$
  }

  \vspace{0.25cm}

  \begin{tikzpicture}
    \fill[navydark, rounded corners=3pt] (0,0) rectangle (10.5,0.45);
    \node[white, font=\footnotesize\bfseries, anchor=west] at (0.2,0.225)
      {\faTrophy\quad Final Conclusion};
  \end{tikzpicture}

  \vspace{0.15cm}
  \begin{center}
  {\footnotesize
  \textbf{WTA is in NP} (solution verifiable in polynomial time) $+$ \textbf{WTA is NP-hard}\\[4pt]
  {\color{goldaccent}$\Rightarrow$} \textbf{The Weapon-Target Assignment Problem is NP-complete!}
  }
  \end{center}

\end{frame}

\begin{frame}{MMR Algorithm --- Overview}
  {\large\bfseries\color{navydark} Maximum Marginal Return (Greedy)}\par
  \vspace{3pt}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{enumerate}\itemsep10pt
        \item \textbf{Scan all pairs}
          \par {\small For every unallocated weapon $w$ and every target $t$:}
          \par {\small\colorbox{navymidHex}{\color{teallt}
            $\;\text{decrease} = V[t] \times p[w][t]\;$}}
        \item \textbf{Pick best pair}
          \par {\small Select weapon-target with \emph{maximum} decrease}
          {\small\color{mygray} ``most bang for the buck''}
        \item \textbf{Assign \& update}
          \par {\small Assign weapon $\to$ target.\quad
            $V[t] \mathrel{-}= \text{maxDecrease}$}
          {\small Remove weapon from pool.}
        \item \textbf{Repeat}
          \par {\small Return to step~1 until all weapons assigned.}
      \end{enumerate}
    \end{column}
    \begin{column}{0.42\textwidth}
      \colorbox{offwhiteHex}{\begin{minipage}{\linewidth}
        \vspace{6pt}
        {\small\bfseries\color{teal} Key Properties}\\[6pt]
        {\small\color{navydark}\bfseries Complexity:\;}
          {\small $O(W^2 \times T)$}\\[5pt]
        {\small\color{navydark}\bfseries Deterministic:\;}
          {\small same input $\to$ same output}\\[5pt]
        {\small\color{navydark}\bfseries Optimality:\;}
          {\small local optimum only}\\[5pt]
        {\small\color{navydark}\bfseries Speed:\;}
          {\small very fast, single pass}
        \vspace{6pt}
      \end{minipage}}
      \vspace{16pt}
      \colorbox{navymidHex}{\begin{minipage}{\linewidth}
        \vspace{5pt}
        {\color{teallt}\small\itshape
         Greedy commits to the locally best choice each round ---
         fast but may miss the globally optimal solution.}
        \vspace{5pt}
      \end{minipage}}
    \end{column}
  \end{columns}
\end{frame}

% ── SLIDE: MMR Pseudocode ─────────────────────────────────────────────
\begin{frame}[shrink=20]{MMR Algorithm --- Pseudocode}
  \begin{center}
  {\footnotesize
  \begin{algorithm}[H]
    \caption{MMR Algorithm}
    \begin{algorithmic}[1]
      \State $\textit{solution.Allocations} \leftarrow \{\}$
      \State $\textit{solution.Value} \leftarrow \textsc{MaxValue}$
      \State $\textit{allocatedWeaponCount} \leftarrow 0$
      \While{$\textit{allocatedWeaponCount} < \textit{numOfWeapons}$}
        \State $\textit{maxDecrease} \leftarrow 0$;\quad $k \leftarrow 1$
        \While{$k \leq \textit{numOfTargets}$}
          \State $\textit{decrease} \leftarrow \textit{solution.Value}[k] \times \textit{KillProbabilities}[\textit{unallocatedWeapon}][k]$
          \If{$\textit{decrease} > \textit{maxDecrease}$}
            \State $\textit{maxDecrease} \leftarrow \textit{decrease}$
            \State $\textit{allocatedTarget} \leftarrow k$
          \EndIf
          \State $k \leftarrow k + 1$
        \EndWhile
        \State $\textit{unallocatedWeapons}.\textsc{Remove}(\textit{allocatedWeapon})$
        \State $\textit{solution.Allocation}[\textit{allocatedWeapon}] \leftarrow \textit{allocatedTarget}$
        \State $\textit{solution.Value}[\textit{allocatedTarget}] \mathrel{-}= \textit{maxDecrease}$
        \State $\textit{allocatedWeaponCount} \leftarrow \textit{allocatedWeaponCount} + 1$
      \EndWhile
      \State $\textit{solution.value} \leftarrow \textsc{CalculateSolutionValue}(\textit{solution},\,\textit{allocations})$
      \State \Return $\textit{solution}$
    \end{algorithmic}
  \end{algorithm}
  }
  \end{center}
\end{frame}

% ── SLIDE: GA Pseudocode ──────────────────────────────────────────────
\begin{frame}[shrink=20]{Genetic Algorithm --- Pseudocode}
  \begin{center}
  {\footnotesize
  \begin{algorithm}[H]
    \caption{Genetic Algorithm}
    \begin{algorithmic}[1]
      \State $\textit{startTime} \leftarrow \textsc{Now}$
      \State $\textit{endTime} \leftarrow \textit{startTime} + \textit{allowedSearchTime}$
      \State $\textit{solution.Allocations} \leftarrow \{\}$
      \State $\textit{solution.value} \leftarrow \textsc{MaxValue}$
      \If{$\textit{noOfTargets} > \textit{noOfWeapons}$}
        \State $\textit{noOfIndividuals} \leftarrow \textit{noOfTargets}$
      \Else
        \State $\textit{noOfIndividuals} \leftarrow \textit{noOfWeapons}$
      \EndIf
      \State $\textit{population} \leftarrow \textsc{GenerateInitialPopulation}(\textit{noOfIndividuals})$
      \State $\textit{individualNo} \leftarrow 1$
      \While{$\textsc{CurrentTime} < \textit{endTime}$}
        \State $\textit{indFromIndiv} \leftarrow \textit{population}[\textit{individualNo}]$
        \State $\textit{indValue} \leftarrow \textsc{CalculateSolutionValue}(\textit{indFromIndiv})$
        \If{$\textit{indValue} < \textit{solution.value}$}
          \State $\textit{solution.value} \leftarrow \textit{indValue}$;\quad $\textit{solution} \leftarrow \textit{indFromIndiv}$
        \EndIf
        \State $\textit{individualNo} \leftarrow \textit{individualNo} + 1$
        \If{$\textit{individualNo} > \textit{noOfIndividuals}$}
          \State $\textit{parents} \leftarrow \textsc{SelectFromIndiv}(\textit{population})$
          \State $\textit{offspring} \leftarrow \textsc{CrossOver}(\textit{parents})$
          \State $\textit{offspring} \leftarrow \textsc{Mutate}(\textit{offspring})$
          \State $\textit{population} \leftarrow \textit{offspring}$;\quad $\textit{individualNo} \leftarrow 1$
        \EndIf
      \EndWhile
      \State \Return $\textit{solution}$
    \end{algorithmic}
  \end{algorithm}
  }
  \end{center}
\end{frame}

\begin{frame}[shrink=5]{Modification 1: MMR-IR (Iterative Refinement)}

  \begin{block}{Overview: Three Enhancements to MMR}
    \small
    The greedy MMR solution is a good starting point but may be suboptimal
    because early decisions cannot be reversed. Our modifications add
    \textbf{post-processing local search} to improve the greedy solution.
  \end{block}

  \vspace{0.2cm}

  \begin{center}
  \begin{tikzpicture}[scale=0.72, every node/.style={font=\footnotesize}]

    \small
    When two weapon-target pairs have \textbf{equal marginal return}, prefer
    the pair targeting the \textbf{higher-value target}.
    This prioritises coverage of critical targets.
  \end{exampleblock}

\begin{frame}[shrink=3]{Modification 2: Hybrid GA --- Overview}

  \begin{block}{Three Weaknesses of Standard GA on WTA}
    \small
    \begin{enumerate}\setlength\itemsep{3pt}
      \item \textbf{Random initialization} --- in high-dimensional space
            ($W \times T$ variables), random solutions are far from optimal
      \item \textbf{No elitism} --- the best solution found can be \textbf{lost}
            when replaced by offspring
      \item \textbf{Blind mutation} --- random target reassignment ignores which
            targets \textbf{need} more weapons
    \end{enumerate}
  \end{block}

  \vspace{0.2cm}

  \begin{alertblock}{Our Three Fixes}
    \vspace{3pt}
    \begin{center}
    \small
    \setlength{\tabcolsep}{9pt}
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{lll}
      \toprule
      \textbf{Weakness}        & \textbf{Our Fix}        & \textbf{Type}         \\
      \midrule
      Random initialisation    & MMR-seeded init         & Warm start            \\
      No elitism / stagnation  & Elitism + detection     & Preservation          \\
      Blind mutation           & Threat-proportional     & \textbf{WTA-specific} \\
      \bottomrule
    \end{tabular}
    \end{center}
    \vspace{2pt}
  \end{alertblock}

\end{frame}

%  Hybrid GA — Mod 1: MMR-Seeded Init

\begin{frame}[shrink=5]{Hybrid GA Mod 1: MMR-Seeded Initialisation}

  \begin{block}{Idea: Start from a Good Solution}
    \small
    Instead of initialising \textbf{all} individuals randomly, seed
    \texttt{population[0]} with the MMR greedy solution.
    The rest remain random for diversity.
  \end{block}

  \vspace{0.15cm}

  \begin{center}
  \begin{tikzpicture}[scale=0.68, every node/.style={font=\footnotesize}]

    % Standard GA row
    \node[font=\bfseries\footnotesize, anchor=east] at (-0.2, 3.05) {Standard GA:};

      \draw[fill=navylight!30, rounded corners=2pt]
        (\i*1.4, 2.7) rectangle (\i*1.4+1.1, 3.3);
      \node at (\i*1.4+0.55, 3.0) {rand};
    }

    % Hybrid GA row
    \node[font=\bfseries\footnotesize, anchor=east] at (-0.2, 1.05) {Hybrid GA:};
    \draw[fill=tealaccent!50, rounded corners=2pt]
      (0, 0.7) rectangle (1.1, 1.3);
    \node[font=\footnotesize\bfseries] at (0.55, 1.0) {MMR};
    \foreach \i in {1,...,4} {
      \draw[fill=navylight!30, rounded corners=2pt]
        (\i*1.4, 0.7) rectangle (\i*1.4+1.1, 1.3);
      \node at (\i*1.4+0.55, 1.0) {rand};
    }

    \draw[->, thick, tealaccent]
      (0.55, 1.3) -- (0.55, 1.9)
      node[right=2pt, font=\scriptsize] {good start!};

  \end{tikzpicture}
  \end{center}

  \vspace{0.12cm}

  \begin{exampleblock}{Why This Works}
    \small
    \begin{itemize}\setlength\itemsep{2pt}
      \item[\color{goldaccent}$\blacktriangleright$]
        MMR gives a \textbf{near-optimal} starting point in $O(W^2T)$ time
      \item[\color{goldaccent}$\blacktriangleright$]
        GA then \textbf{refines} this via crossover and mutation
      \item[\color{goldaccent}$\blacktriangleright$]
        Random individuals provide \textbf{genetic diversity} to avoid local optima
      \item[\color{goldaccent}$\blacktriangleright$]
        For large instances ($W=375$), this saves hundreds of wasted generations
    \end{itemize}
  \end{exampleblock}

\end{frame}


% ─────────────────────────────────────────────────────────────────────────────
%  Hybrid GA — Mod 2: Elitism + Stagnation
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}[shrink=5]{Hybrid GA Mod 2: Elitism + Stagnation Detection}

  \begin{block}{Idea: Never Lose the Best --- Re-diversify When Stuck}
    \small
    \begin{itemize}\setlength\itemsep{3pt}
      \item[\color{goldaccent}$\blacktriangleright$]
        \textbf{Elitism:} Preserve top 10\,\% of population each generation
      \item[\color{goldaccent}$\blacktriangleright$]
        \textbf{Stagnation detection:} If best fitness is unchanged for
        5 consecutive generations, inject fresh random individuals to re-diversify
    \end{itemize}
  \end{block}

  \vspace{0.15cm}

  \begin{center}
  \begin{tikzpicture}[scale=0.68, every node/.style={font=\footnotesize}]

    \node[draw, rounded corners=3pt, fill=navymid!60, text=white,
          minimum width=3.2cm, minimum height=0.75cm] (gen1) at (0, 3)
      {Generation $g$};

    \node[draw, rounded corners=3pt, fill=navymid!60, text=white,
          minimum width=3.2cm, minimum height=0.75cm] (gen2) at (0, 0)
      {Generation $g+1$};

    \draw[->, very thick, tealaccent]
      (-2.0, 3) -- (-2.0, 0)
      node[midway, left=2pt] {Top 10\%};

    \draw[->, thick]
      (2.0, 3) -- (2.0, 0)
      node[midway, right=2pt] {Offspring (90\%)};

    \node[draw, dashed, rounded corners=3pt, fill=goldaccent!20,
          text width=3.6cm, align=center, font=\scriptsize] (stag) at (7.2, 1.5)
      {If stagnant $\geq$ 5 gens:\\[1pt]inject random individuals};

    \draw[->, thick, goldaccent, dashed] (stag.west) -- (2.6, 0.4);

  \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}

  \begin{alertblock}{Without Elitism (Standard GA)}
    \small
    A good solution in generation $g$ can be completely lost if all offspring
    are worse --- especially problematic with small time budgets.
  \end{alertblock}

\end{frame}


% ─────────────────────────────────────────────────────────────────────────────
%  Hybrid GA — Mod 3: Threat-Proportional Mutation
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}[shrink=5]{Hybrid GA Mod 3: Threat-Proportional Mutation}

  \begin{block}{Idea: Domain-Aware Mutation (Novel, WTA-Specific)}
    \small
    Instead of mutating to a \textbf{uniformly random} target, bias mutation
    toward targets with \textbf{high remaining threat}:
    \[
      \text{remaining\_threat}[j]
        = V_j \times \prod_{i:\,x_i = j} (1 - p_{ij})
    \]
    \textbf{Mutation target selection:}\quad
    $P(\text{mutate to } j) \propto \text{remaining\_threat}[j]$
  \end{block}

  \vspace{0.15cm}

  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Standard Mutation}
        \small
        \begin{itemize}\setlength\itemsep{2pt}
          \item[\color{goldaccent}$\blacktriangleright$]
            Pick random target uniformly
          \item[\color{goldaccent}$\blacktriangleright$]
            May reassign to an already well-covered target
          \item[\color{goldaccent}$\blacktriangleright$]
            No domain knowledge used
        \end{itemize}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Threat-Proportional Mutation}
        \small
        \begin{itemize}\setlength\itemsep{2pt}
          \item[\color{goldaccent}$\blacktriangleright$]
            High-value, poorly-covered targets attract more mutations
          \item[\color{goldaccent}$\blacktriangleright$]
            Focuses search where improvement is \textbf{most needed}
          \item[\color{goldaccent}$\blacktriangleright$]
            \textbf{Novel:} Our original WTA-specific contribution
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}

  \vspace{0.15cm}

  \begin{block}{Concrete Example}
    \small
    If T1 ($V=100$, survival$=0.6$) and T2 ($V=20$, survival$=0.1$):\\[3pt]
    \quad threat$_1 = 100 \times 0.6 = 60$,\qquad
         threat$_2 = 20 \times 0.1 = 2$\\[3pt]
    $\Rightarrow$ T1 is $30\times$ more likely to receive a mutated weapon than T2
  \end{block}

\end{frame}
